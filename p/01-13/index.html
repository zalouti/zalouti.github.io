这是mkdir.py

import os

def mkdir(path):
    # 去除首位空格
    path = path.strip()
    # 去除尾部 \ 符号
    path = path.rstrip("\\")
    # 判断路径是否存在
    # 存在     True
    # 不存在   False
    isExists = os.path.exists(path)

    # 判断结果
    if not isExists:
        # 如果不存在则创建目录
        # 创建目录操作函数
        os.makedirs(path)
        print(path+' 创建成功')
        return True
    else:
        # 如果目录存在则不创建，并提示目录已存在
        print(path+' 目录已存在')
        return False

计算部分(fundamental solution)

开始正式的计算，但是也分几步

  1. 首先划分网格，使用Gaussian积分，为每一个格点匹配对应的权重
  2. 之后再在每一个网格上积分——一个求和——但注意是平面积分——求和两次

网格划分mesh_div.py包括了根据读取的数据Nx和Ny划分网格mesh，以及每个区间做区间变换interval_tran

区间变换为： $$ \int _{a}^{b}f(x)\ dx=\frac{b-a}{2}\int _{-1}^{1}f\left({\frac {b-a}{2}}\xi +{ \frac {a+b}{2}}\right)\ d\xi $$

import numpy as np

def mesh(Lx, Ly, Nx, Ny):
    x = np.linspace(0, Lx, Nx+1)
    y = np.linspace(0, Ly, Ny+1)
    # 划分Nx个网格，需要有Nx+1个点
    return x, y
# boundary Lx ,Ly
# Number of meshes Nx, Ny

def interval_tran(start, end, n):
    p, w = np.polynomial.legendre.leggauss(n)
    coff = (end-start)/2
    for i in range(n):
        p[i] = ((end-start)*p[i]+start+end)/2
    return p, w, coff
# transform gaussian points from [-1,1] to [start,end]
# n stands for the num of Gaussian points
# start&end stand for the start and end of the interval
# return the guassian points on new interval with their weights

积分部分intergrate.py首先写一个非常一般的积分，即高斯求积法在$[-1,1]$上的积分 $$ \int_{-1}^1f(x)dx \approx \sum_{i=1}^nw_if_(x_i) $$

def int_basic(n, point, weight, func1):
    result = 0
    for i in range(n):
        result = result+func1(point[i])*weight[i]  
    return result

最后得到一维的高斯求积法为

def int_1D(start, end, n, func1):
    point, weight, coff = interval_tran(start, end, n)
    result = coff*int_basic(n, point, weight, func1)
    return result

考虑到区间变换，则二维的高斯求积法可以被写作 $$ \int_{x_1}^{x_2}\int_{y_1}^{y_2}f(x,y)dydx =\\ \\ \frac{x_2-x_1}{2}\frac{y_2-y_1}{2}\int^1_{-1}\int_{-1}^1f(\frac{x_2-x_1}{2}x+\frac{x_2+x_1}{2},\frac{y_2-y_1}{2}y+\frac{y_2+y_1}{2})\\ \approx \frac{x_2-x_1}{2}\frac{y_2-y_1}{2}\sum_{j=1}^m\sum_{i=1}^nw_iw_jf(x_i,y_j) $$ 我们可以看到，上述二维的表达式中，最后一项两个求和可以采用向量化编程的形式，减少for循环的出现，即为： $$ \sum_{j=1}^m\sum_{i=1}^nw_iw_jf(x_i,y_j)= \begin{pmatrix} w_1&w_2&\cdots&w_n \end{pmatrix} \begin{pmatrix} f(x_1,y_1)&\cdots&f(x_1,y_m)\\ \vdots&&\vdots\\ f(x_n,y_1)&\cdots&f(x_n,y_m) \end{pmatrix} \begin{pmatrix} w_1\\ w_2\\ w_m \end{pmatrix}\\ =det\begin{pmatrix} f(x_1,y_1)&\cdots&f(x_1,y_m)\\ \vdots&&\vdots\\ f(x_n,y_1)&\cdots&f(x_n,y_m) \end{pmatrix} \begin{pmatrix} w_1\\ w_2\\ w_m \end{pmatrix} \begin{pmatrix} w_1&w_2&\cdots&w_n \end{pmatrix} $$ 代码实现为：

def int_2D(x1,x2,y1,y2,n,func2):
    x_point, x_weight, x_coff = interval_tran(x1, x2, n)
    y_point, y_weight, y_coff = interval_tran(y1, y2, n)
    w2d=np.outer(y_weight,x_weight)
    p2d=np.meshgrid(x_point,y_point)
    result=x_coff*y_coff*func2(p2d)*w2d
    return result.sum()
