Milestone One: Form k-Grams

分辨不同语言的一种方法便是提取字符串中的子串，统计不同子串的频率。

第一个里程碑，题目要求提取出一个字符串中的k长度的子串并统计频率，采用Map即可：

Map<string, double> kGramsIn(const string& str, int kGramLength) {
    Map<string, double> kGram;
    if (kGramLength<=0)
        error("kGramLength should be positive!");
    else {
        if (str.length() < kGramLength)
            return Map<string, double>{};
        else
        {
            int cnt = 0;
            while(kGramLength + cnt <= str.length())
            {
               kGram[str.substr(cnt,kGramLength)]+=1;
               cnt+=1;
            }
        }
    }
   return kGram;
}

注意针对这个函数的测试样例只有前八个，报错是因为后面的函数还没有写hhh。

Milestone Two: Normalize Frequencies

针对不同的文本，子串的频率会发生变化，所以我们需要对其归一化处理。

题目中给出的方法是，先求所有数据平方和后开方，之后每个数据除以此平方根即可。

Map<string, double> normalize(const Map<string, double>& input) {
    Map<string, double> NormMap;
    if (input.isEmpty())
        error("The Map is empty!");
    else
    {
        double sqrtsum = 0;
        for(string str: input)
        {
            sqrtsum += input[str]*input[str];
        }
        sqrtsum = sqrt(sqrtsum);
        if (sqrtsum - 0 < 10e-6)
            error("The Map is Allzero!");
        for(string str: input)
        {
            NormMap[str] = input.get(str)/sqrtsum;
        }
    }
    return NormMap;
}

Milestone Three: Filter Out Uncommon Trigrams

当一些特殊文本出现了一些特殊的词语时，我们应该过滤掉他们，这就是这一部分所做的工作。题目限制使用了PriorityQueue类型。

#include "priorityqueue.h" //Remember to add this line.
Map<string, double> topKGramsIn(const Map<string, double>& source, int numToKeep) {
    Map<string, double> mapkeeped = source;
    PriorityQueue<string> sourcecopy;
    if (numToKeep < 0)
        error("numToKeep shouldn't be negtive!");
    else if (numToKeep == 0)
        return Map<string, double> {};
    else
    {
        if (numToKeep >= source.size())
            return source;
        else
        {
            for (string str: source)
                sourcecopy.enqueue(str,source[str]);
            while(sourcecopy.size() > numToKeep)
                mapkeeped.remove(sourcecopy.dequeue());
        }
    }
    return mapkeeped;
}

Milestone Four: Implement Cosine Similarity

比对两组文本的相似度，我们采用计算Cosine Similarity的方法。即，相同的子串对应的概率相乘，最后求和即可。

double cosineSimilarityOf(const Map<string, double>& lhs, const Map<string, double>& rhs) {
    double cossim = 0;
    if(lhs.equals(rhs))
        return 1;
    else
    {
        for(string str: lhs)
        {
            if(rhs.containsKey(str))
                cossim += lhs[str]*rhs[str];
        }
    }
    return cossim;
}

Milestone Five: Guess a Textʼs Language

终于到了最后的时刻了，利用给定的语料库Set，找出最高的余弦相似度即可。

string guessLanguageOf(const Map<string, double>& textProfile,
                       const Set<Corpus>& corpora) {
    Map<string, double> languagemap;
    double possibility = 0;
    if (corpora.isEmpty())
        error("Set corpora is Empty!");
    else
    {
        for (Corpus lang :corpora)
        {
            if(cosineSimilarityOf(textProfile,lang.profile) > possibility)
            {
                possibility = cosineSimilarityOf(textProfile,lang.profile);
                languagemap.clear();
                languagemap.put(lang.name,possibility);
            }
        }
    }
    return languagemap.firstKey();
}

Part Two: Rising Tides

这道题目是广度优先搜索(Breadth-First Search)的一道训练题目，背景是洪水来袭后的淹没检测。

这道题目还是花了不少时间的，看来题目长度和难度不成正比……

首先梳理一下核心的思路：

  • 首先判断最初的位置是否被淹没，即第0步的位置，将第零步加入queue中
  • 之后从第零步出发（即从queue中拿出一个），向周围的四个方向淹没，将被淹没的地方（第二步）加入queue中
  • 依此类推至第X步，直到最后queue中没有对象时，结束搜索

老师非常贴心的给出了伪代码实现，所以按照伪代码翻译出来即可。

Vector<GridLocation> adjacent(const GridLocation& center)
{
    GridLocation Left,Right,Up,Down;
    Left.row = center.row;
    Right.row = center.row;
    Up.col = center.col;
    Down.col = center.col;
    Right.col = center.col - 1;
    Left.col = center.col + 1;
    Up.row = center.row - 1;
    Down.row = center.row + 1;
    return Vector<GridLocation> {Left,Right,Up,Down};
}

Grid<bool> floodedRegionsIn(const Grid<double>& terrain,
                            const Vector<GridLocation>& sources,
                            double height) {
    Grid<bool> bloodregion;
    bloodregion.resize(terrain.numRows(),terrain.numCols());
    //    create an empty queue;
    Queue<GridLocation> newterrain ;
    //    for (each water source at or below the water level)
    for (GridLocation watersource: sources)
    {
        //        flood that square;
        if(terrain.inBounds(watersource) && height >= terrain[watersource.row][watersource.col])
        {
            bloodregion[watersource.row][watersource.col] = 1;
            //        add that square to the queue;
            newterrain.enqueue(watersource);
        }
    }
    GridLocation waternow;
    //    while (the queue is not empty)
    while (!newterrain.isEmpty()) {
        //        dequeue a position from the front of the queue;
        waternow = newterrain.dequeue();
        //        for (each square adjacent to the position in a cardinal direction)
        for (GridLocation surround: adjacent(waternow))
        {
            //  if (that square is at or below the water level and isn't yet flooded)
            if(terrain.inBounds(surround) && terrain[surround.row][surround.col] <= height
                    && bloodregion[surround.row][surround.col] != 1)
            {    //          flood that square;
                bloodregion[surround.row][surround.col] = 1;
                //           add that square to the queue;
                newterrain.enqueue(surround);
            }
        }
    }
    return bloodregion;
}
