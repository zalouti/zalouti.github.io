Part Two: Only Connect

题目背景是BBC的一些文字游戏，要求采用递归的方法解决问题，提示给得相当充分，写起来不困难。

	string onlyConnectize(string phrase) {
    /* TODO: The next few lines just exist to make sure you don't get compiler warning messages
     * when this function isn't implemented. Delete these lines, then implement this function.
     */
    if (phrase=="")
        return "";
    else if(phrase.length()==1)
    {
        if(isalpha(phrase[0]))
        {
            char c = toupper(phrase[0]);
            if(c!='A' && c!='E' && c!='I' && c!='O' && c!= 'U')
                return toUpperCase(phrase);
        }
        return "";
    }
    return onlyConnectize(phrase.substr(0,1))+onlyConnectize(phrase.substr(1));
}

Part Four: Sandpiles

题目背景是往沙盘中扔谷子，每满四则向四周溢出，原格点位置清零。

void dropSandOn(Grid<int>& world, int row, int col) {
    if(world.inBounds(row,col))
    {
        if((world[row][col]+1)<=3)
            world[row][col]+=1;
        else
        {
            world[row][col]=0;
            dropSandOn(world,row-1,col);
            dropSandOn(world,row+1,col);
            dropSandOn(world,row,col-1);
            dropSandOn(world,row,col+1);
        }
    }
}

Part Five: Plotter

一道有趣的题目，老师已经搭好了图形化的其他框架，只需要完成绘制部分的操作即可。这里我采用了作业中提示的方法，利用istream读取每一行之后逐行分割。容易疏忽的地方在于，虽然处于PenUp的时候不能绘制，但是画笔是可以移动的。

搞清楚这一点之后合理分层级就好，注意此题并不需要使用递归。

#include "Plotter.h"
#include "strlib.h"
#include <vector.h>
using namespace std;


void runPlotterScript(istream& input) {
    PenStyle style = { 1, "black" };
    double x1=0,x2=0;
    int flag=0;
    for (string line; getline(input, line); ) {
        Vector<string> parameter = stringSplit(line, " ");
        if(equalsIgnoreCase(parameter[0],"PenDown"))
            flag=1;
        else if(equalsIgnoreCase(parameter[0],"PenUp"))
            flag=0;
        else if(equalsIgnoreCase(parameter[0],"MoveAbs"))
        {
            if(flag==1)
                drawLine(x1,x2,stringToReal(parameter[1]),stringToReal(parameter[2]),style);
            x1=stringToReal(parameter[1]);
            x2=stringToReal(parameter[2]);
        }
        else if(equalsIgnoreCase(parameter[0],"MoveRel"))
        {
            if(flag==1)
                drawLine(x1,x2,x1+stringToReal(parameter[1]),x2+stringToReal(parameter[2]),style);
            x1+=stringToReal(parameter[1]);
            x2+=stringToReal(parameter[2]);
        }
        else if(equalsIgnoreCase(parameter[0],"PenColor"))
            style.color=parameter[1];
        else if(equalsIgnoreCase(parameter[0],"PenWidth"))
            style.width=stringToReal(parameter[1]);
    }
}
